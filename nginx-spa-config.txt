# NGINX Konfiguration für SPA-Routing (React App)
# Diese Konfiguration muss in die nginx server {} Block eingefügt werden

server {
    listen 80;
    listen 443 ssl;
    server_name www.kiramariecremer.de kiramariecremer.de;
    
    root /pfad/zur/website/public_html;
    index index.html index.htm;
    
    # Gzip Kompression
    gzip on;
    gzip_vary on;
    gzip_min_length 1024;
    gzip_types text/plain text/css text/xml text/javascript application/javascript application/xml+rss application/json;
    
    # Asset-Caching
    location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg|webp|woff|woff2|ttf|otf)$ {
        expires 1y;
        add_header Cache-Control "public, immutable";
        try_files $uri =404;
    }
    
    # Font MIME-Types
    location ~* \.(woff|woff2)$ {
        add_header Access-Control-Allow-Origin *;
        expires 1y;
        add_header Cache-Control "public, immutable";
    }
    
    # API und Admin-Pfade direkt weiterleiten
    location ^~ /admin/ {
        try_files $uri $uri/ =404;
    }
    
    location ^~ /api/ {
        try_files $uri $uri/ =404;
    }
    
    location ^~ /webhook.php {
        try_files $uri =404;
    }
    
    # Static assets direkt ausliefern
    location ^~ /static/ {
        try_files $uri =404;
    }
    
    location ^~ /uploads/ {
        try_files $uri =404;
    }
    
    location ^~ /images/ {
        try_files $uri =404;
    }
    
    location ^~ /fonts/ {
        try_files $uri =404;
    }
    
    location ^~ /videos/ {
        try_files $uri =404;
    }
    
    location ^~ /data/ {
        try_files $uri =404;
    }
    
    # SPA-Routing: Alle anderen Requests zu index.html
    location / {
        try_files $uri $uri/ /index.html;
    }
    
    # Error pages
    error_page 404 /index.html;
}

# WICHTIGE SCHRITTE FÜR FLORIAN:
# 1. Diese Konfiguration in /etc/nginx/sites-available/kiramariecremer.de einsetzen
# 2. nginx -t (Syntax check)
# 3. systemctl reload nginx
# 4. /pfad/zur/website/ durch echten Pfad ersetzen
